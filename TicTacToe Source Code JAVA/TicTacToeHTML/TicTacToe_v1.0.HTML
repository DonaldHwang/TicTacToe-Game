01 /**

04 public enum Tool {
05   X, O, EMPTY;   // name()
06   
07   public String toString () { return SHOW_AS[ ordinal() ]; }

01 /**
02    A class for Tic Tac Toe moves
03 */
04 public class Move {
05   public Move (int r, int c) {
06     if (r < 1 || Board.SIZE < r || c < 1 || Board.SIZE < c)
07       throw new IllegalArgumentException();
08 
09     row = r;
10     column = c;
11   }
12 
13   public int getRow ()  { return row; }
14   
15   public int getColumn ()   { return column; }
16 
17   private int row;
18   private int column;
19 }

01 /**
02    A class for storing, manipulating, and printing TicTacToe boards
03 */
04 public class Board {
05   /*
06     A sample board is shown like this:
07   
08       Here is the current board:
09      
10       o| |x
11       -+-+-
12       x|o|o 
13       -+-+-
14        | |x
15    
16   */
17   public void show () {
18     System.out.println( "Here is the current board:\n" );
19 
20     for (int r = 1; r <= SIZE; r++) {
21       for (int c = 1; c <= SIZE; c++) {
22         System.out.print( board[r-1][c-1] );  
23         if (c != SIZE)   // Print strut after all but last column
24           System.out.print( "|" );
25       }
26       System.out.println();     
27 
28       if (r != SIZE)   // Print row line after all but last row
29         System.out.println( "-+-+-" );
30     }
31     System.out.println();     
32   }
33     
34   public boolean isGameWon () {
35     final Tool[][] b = board;  // a local variable for shorter expressions
36 
37     // Check (short circuit) all rows, columns and diagonals for a win
38     return
39       b[0][0]!=Tool.EMPTY && b[0][0]==b[0][1] && b[0][1]==b[0][2] ||  // Row 0
40       b[1][0]!=Tool.EMPTY && b[1][0]==b[1][1] && b[1][1]==b[1][2] ||  // Row 1
41       b[2][0]!=Tool.EMPTY && b[2][0]==b[2][1] && b[2][1]==b[2][2] ||  // Row 2
42                                                                       
43       b[0][0]!=Tool.EMPTY && b[0][0]==b[1][0] && b[1][0]==b[2][0] ||  // Col 0
44       b[0][1]!=Tool.EMPTY && b[0][1]==b[1][1] && b[1][1]==b[2][1] ||  // Col 1
45       b[0][2]!=Tool.EMPTY && b[0][2]==b[1][2] && b[1][2]==b[2][2] ||  // Col 2
46                                                                       
47       b[1][1]!=Tool.EMPTY && b[0][0]==b[1][1] && b[1][1]==b[2][2] ||  // Dia 1
48       b[1][1]!=Tool.EMPTY && b[2][0]==b[1][1] && b[1][1]==b[0][2] ;   // Dia 2
49   }
50 
51   public boolean isFull () {
52     for (int i = 0; i < SIZE; i++)
53       for (int j = 0; j < SIZE; j++)
54         if (board[i][j] == Tool.EMPTY) return false;
55     return true;
56   }
57     
58   public boolean isValid (Move move) {
59     int r = move.getRow();
60     int c = move.getColumn();
61     return board[r-1][c-1] == Tool.EMPTY;
62   }
63     
64   public void handleMove (Move move, Tool player) {
65     int r = move.getRow();
66     int c = move.getColumn();
67     System.out.printf( "\nThe move for %s is %d, %d\n", player, r, c );
68 
69     board[r-1][c-1] = player;  // Place the player's Tool on the board
70   }
71 
72   public void clear () { 
73     for (int i = 0; i < SIZE; i++)
74       for (int j = 0; j < SIZE; j++)
75         board[i][j] = Tool.EMPTY;
76   }
77 
78   public static final int SIZE = 3;
79   private Tool[][] board = new Tool[SIZE][SIZE];
80 }

001 /**
002    A class playing a game of TicTacToe
003 */
004 import java.util.Random;
005 import java.util.Scanner;
006 
007 public class TicTacToe {
008   public void play () {
009     showStartHint();
010     randomFirstPlayer();  
011 
012     board.clear();
013     board.show();
014 
015     boolean gameOver = false;
016     while (!gameOver) {
017       Move move = getAMove();
018       board.handleMove( move, player );
019 
020       board.show();
021 
022       if (board.isGameWon() || board.isFull())
023         gameOver = true;
024       else 
025         player = oppositePlayer();
026     }
027 
028     showGameResult();
029   }
030 
031   private void showStartHint() {
032     System.out.println( HINT_MESSAGE);
033   }
034 
035   private void randomFirstPlayer () {
036     if (generator.nextBoolean()) {
037       person   = Tool.X; 
038       computer = Tool.O;
039     } else {
040       person   = Tool.O; 
041       computer = Tool.X;
042     }
043     player = Tool.X;
044   }
045     
046   private Tool oppositePlayer () {
047     return (player == computer) ? person : computer;
048   }
049 
050   private void showGameResult () {
051     if (board.isGameWon())
052       System.out.println( player==person ? "You won!" : "I won!" );
053     else if (board.isFull())
054       System.out.println( "We tied!" );
055     else
056       System.out.println( "Something went wrong!" );
057   }
058   
059    // Creates a move, either a random generated move or as input from the user
060   private Move getAMove () {
061     Move move = null;
062 
063     if (player == computer) {
064       System.out.println( "It is my move.  I am '" + player + "'" );
065       move = randomGenerateAValidMove();
066     } else {
067       System.out.println( "It is your move.  You are '" + player + "'" );
068       move = getAValidMoveFromPerson();
069     }
070     return move;
071   }
072 
073   private Move randomGenerateAValidMove () {
074     Move move = null;
075     do {
076       move = new Move( intBetween(1, Board.SIZE), intBetween(1, Board.SIZE));
077     } while ( !board.isValid( move ) );
078     return move;
079   }
080 
081   private int intBetween (int low, int high) {
082     return low + generator.nextInt( high - low + 1 );
083   }
084 
085   private Move getAValidMoveFromPerson () {
086     Move move = null;
087     while (true) {
088       try {
089         System.out.print( "Enter a row and column on one line: " );
090         move = new Move( in.nextInt(), in.nextInt() );
091         // Generates an exception if can't make a move from r and c
092 
093         if (board.isValid( move )) return move;
094 
095         System.out.println( "Invalid move. Try again!" );
096       } catch (Exception e) {
097         System.out.println( "Input error. Try again!" );
098       }
099     }
100   }

102   private Tool player;
103   private Tool computer;
104   private Tool person;
105 
106   private Board board = new Board();
107   private Scanner in = new Scanner( System.in );
108 
109   private static Random generator = new Random();
110 
111   private static final String HINT_MESSAGE = "\n" +
112     "************************************************\n" +
113     "Let's play Tic Tac Toe!\n" +
114     "When asked for a move, enter location you want.\n" +
115     "Enter the row first and then the column, both on the same line.\n" +
116     "The row and column must in the range 1 .. 3\n" +
117     "************************************************\n" ;
118 }
